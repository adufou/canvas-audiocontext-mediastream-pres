Canvas API

The 2D rendering context — drawing images, managing state with a stack, and clipping regions.

1

drawImage

ctx.drawImage() accepts an HTMLImageElement, HTMLCanvasElement, or HTMLVideoElement (which we'll use in the MediaStream demo).

2

save / restore — the state stack

ctx.save() pushes the current state (transforms, styles, clip) onto a stack. ctx.restore() pops it. Transforms applied inside a save/restore block don't leak out.

Stack depth: 0
3

clip — masking with a path

After calling ctx.clip(), all subsequent drawing is masked to the current path. Wrap in save() / restore() to remove the clip.

4

Combined — animated drawImage + clip + save/restore

Putting it all together: an image drawn each frame, clipped to a circle, rotating inside a save()/restore() block.

Connected to MediaStream This canvas can become a live video stream via canvas.captureStream(30) — see the MediaStream demo →
canvas.html JS 
// Step 1 — drawImage
const canvas = document.getElementById('c1');
const ctx = canvas.getContext('2d');

const img = new Image();
img.src = 'data:image/svg+xml,...';

img.onload = () => {
  // Draw at (x, y) with width, height
  ctx.drawImage(img, 50, 25, 300, 200);
};

// drawImage also accepts:
// - HTMLCanvasElement  (draw one canvas into another)
// - HTMLVideoElement   (draw a video frame — see MediaStream demo)
// - ImageBitmap, OffscreenCanvas, ...
// Step 2 — save / restore state stack
const ctx = canvas.getContext('2d');

// save() pushes the current state:
//   - current transform matrix
//   - current clipping region
//   - current strokeStyle, fillStyle, lineWidth...
ctx.save();            // stack depth: 1

ctx.translate(100, 50);
ctx.rotate(Math.PI / 6);

ctx.fillRect(0, 0, 80, 40);

ctx.restore();           // stack depth: 0 — back to origin

// The rectangle drawn AFTER restore() is unaffected
// by the translate + rotate we applied above.
ctx.fillRect(200, 100, 80, 40);
// Step 3 — clip()
const ctx = canvas.getContext('2d');

// Save before clipping so we can restore later
ctx.save();

// Define a circular clip path
ctx.beginPath();
ctx.arc(200, 150, 120, 0, Math.PI * 2);

// clip() — subsequent drawing is masked to this path
ctx.clip();

// This fill only appears inside the circle
ctx.fillStyle = '#7c6af7';
ctx.fillRect(0, 0, 400, 300);

// restore() removes the clip region
ctx.restore();

// This fill covers the whole canvas
ctx.fillStyle = 'rgba(255,255,255,0.1)';
ctx.fillRect(0, 0, 400, 300);
// Step 4 — Combined animated demo
let angle = 0;
let rafId = null;

function draw(img) {
  const cx = canvas.width / 2;
  const cy = canvas.height / 2;
  const r  = 120;

  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // save() — we will clip and rotate, don't want it to leak
  ctx.save();

  // 1. Define a circular clip path
  ctx.beginPath();
  ctx.arc(cx, cy, r, 0, Math.PI * 2);
  ctx.clip();

  // 2. Move origin to canvas center, apply rotation
  ctx.translate(cx, cy);
  ctx.rotate(angle);

  // 3. Draw image centered on the (now rotated) origin
  ctx.drawImage(img, -r, -r, r * 2, r * 2);

  // restore() — clip and transform are gone
  ctx.restore();

  angle += 0.01;
  rafId = requestAnimationFrame(() => draw(img));
}

requestAnimationFrame(() => draw(img));
